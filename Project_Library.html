 74     public static void BooksMenu()
 75     {//begin BooksMenu
 76         Scanner keyboard = new Scanner(System.in);
 77         String WelcomeMenu;
 78         int MenuOption;
 79 
 80         System.out.println("This is the Books Option Menu.");
 81 
 82         do
 83         {//begin do
 84        System.out.println("Enter the option you want"
 85                 + "to do:\n [1] to Search for books."
 86                 + "\n [2] to list all the books."
 87                 + "\n [3] to Edit the books."
 88                 + "\n [0] to Exit to the main menu");
 89        System.out.print(": ");
 90        WelcomeMenu = keyboard.nextLine();
 91         MenuOption= Integer.parseInt(WelcomeMenu);
 92 
 93         switch (MenuOption)
 94         {//begin switch
 95             case 0: break;
 96             case 1: BookSearch();break;
 97             case 2: AllBooks();break;
 98             case 3: EditBooks();break;
 99             default:System.out.println("Invalid Option"); break;
100         }//end switch
101 
102         }//end do
103         while(MenuOption!=0);
104 
105         System.out.println("End of Books Menu.");
106 
107 
108     }//end BooksMenu
